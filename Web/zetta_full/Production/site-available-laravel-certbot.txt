server { 
    server_name api.talkenoo.click;
    root /var/www/zetta_api/public;
    index index.php index.html index.htm;
    
    # Bloc pour le challenge HTTP-01 de Certbot
    location /.well-known/acme-challenge/ {
        default_type "text/plain";
        root /var/www/certbot;  # Assurez-vous que ce répertoire existe
    }
    
    location / {
        try_files $uri $uri/ /index.php?$query_string;
    }
    
    location ~ \.php$ {
        include snippets/fastcgi-php.conf;
        fastcgi_pass unix:/run/php/php8.3-fpm.sock; # Vérifie ta version PHP
        fastcgi_param SCRIPT_FILENAME $document_root$fastcgi_script_name;
        include fastcgi_params;
    }
    
    location ~ /\.ht {
        deny all;
    }


 location /app {
    proxy_http_version 1.1;
    proxy_set_header Host $http_host;
    proxy_set_header Scheme $scheme;
    proxy_set_header SERVER_PORT $server_port;
    proxy_set_header REMOTE_ADDR $remote_addr;
    proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    proxy_set_header Upgrade $http_upgrade;
    proxy_set_header Connection "Upgrade";

    proxy_pass http://0.0.0.0:8080;
}

    
    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/api.talkenoo.click/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/api.talkenoo.click/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot
}

server {
    if ($host = api.talkenoo.click) {
        return 301 https://$host$request_uri;
    } # managed by Certbot
    
    listen 80;
    server_name api.talkenoo.click www.api.talkenoo.click;
    return 404; # managed by Certbot
}
